321 def get_country(city1, country1, city2, embeddings):
322  # store the city1, country 1, and city 2 in a set called group
323  group = set((city1, country1, city2))
324  # get embeddings of city 1
325  city1_emb = embeddings[city1]
326  # get embedding of country 1
327  country1_emb = embeddings[country1]
328  # get embedding of city 2
329  city2_emb = embeddings[city2]
330  # get embedding of country 2 (it's a combination of the embeddings of country 1, city 1 and city 2)
331  # Remember: King - Man + Woman = Queen
332  vec = city2_emb - city1_emb+country1_emb
333  # Initialize the similarity to -1 (it will be replaced by a similarities that are closer to +1)
334  similarity = -1
335  # initialize country to an empty string
336  country = ''
337  # loop through all words in the embeddings dictionary
338  for word in embeddings.keys():
339  # first check that the word is not already in the 'group'
340  if word not in group:
341  # get the word embedding
342  word_emb = embeddings[word]
343  # calculate cosine similarity between embedding of country 2 and the word in the embeddings dictionary
344  cur_similarity = cosine_similarity(word_emb, vec)
345  # if the cosine similarity is more similar than the previously best similarity...
346  if cur_similarity > similarity:
347  # update the similarity to the new, better similarity
348  similarity = cur_similarity
349  # store the country as a tuple, which contains the word and the similarity
350  country = (word, similarity)
351  return country
